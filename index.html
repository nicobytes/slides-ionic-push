Configuration

						
import { CloudSettings, CloudModule } from '@ionic/cloud-angular';

const cloudSettings: CloudSettings = {
  core: {
    'app_id': 'APP_ID'
  }
};

@NgModule({
  ...
  imports: [
    IonicModule.forRoot(MyApp),
    CloudModule.forRoot(cloudSettings)
  ],
  ...
})
export class AppModule {}

Check src/app/app.module.ts

now, Ionic Push

Prerequisites

  • Cloud Client installed and configured
  • A physical Android or iOS Device
  • A Security Profile with Push Credentials (see FCM Project & API Key and iOS Push Certificate)

Generating a Server key and Sender ID in firebase.google.com

Attach Credentials in apps.ionic.io

Plugin Installation

						
ionic cordova platform add android
ionic cordova plugin add phonegap-plugin-push --variable SENDER_ID=your_sender_id --save

Cloud Configuration

						
const cloudSettings: CloudSettings = {
  'core': {
    'app_id': 'APP_ID',
  },
  'push': {
    'sender_id': 'SENDER_ID',
    'pluginConfig': {
      'ios': {
        badge: true,
        sound: true
      },
      'android': {
        iconColor: '#343434',
        forceShow: true
      }
    }
  }
};

Injecting Push (app.component.ts)

						
import {Component} from '@angular/core';
import {
  Push,
  PushToken
} from '@ionic/cloud-angular';

@Component( ... )
export class MyApp {
  constructor(private push: Push) {
    ...
  }
}

Registering Device Tokens

						
this.push.register().then((t: PushToken) => {
 return this.push.saveToken(t,{
  ignore_user: true
 });
}).then((t: PushToken) => {
 console.log('Token saved:', t.token);
});

Handling Notifications

						
this.push.rx.notification()
  .subscribe((msg) => {
    alert(msg.title + ': ' + msg.text);
  });
